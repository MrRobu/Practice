DROP TABLE IF EXISTS Album CASCADE;
DROP TABLE IF EXISTS AlbumTrack CASCADE;
DROP TABLE IF EXISTS Artist CASCADE;
DROP TABLE IF EXISTS ArtistAlbum CASCADE;
DROP TABLE IF EXISTS ArtistGenre CASCADE;
DROP TABLE IF EXISTS Band CASCADE;
DROP TABLE IF EXISTS BandGenre CASCADE;
DROP TABLE IF EXISTS BandArtist CASCADE;
DROP TABLE IF EXISTS BandAlbum CASCADE;
DROP TABLE IF EXISTS FooUser CASCADE;
DROP TABLE IF EXISTS Genre CASCADE;
DROP TABLE IF EXISTS Playlist CASCADE;
DROP TABLE IF EXISTS PlaylistTrack CASCADE;
DROP TABLE IF EXISTS Track CASCADE;
DROP TABLE IF EXISTS AlbumReview CASCADE;
DROP TABLE IF EXISTS TrackReview CASCADE;
DROP TABLE IF EXISTS BandReview CASCADE;
DROP TABLE IF EXISTS ArtistReview CASCADE;

CREATE TABLE Genre(
	ID SERIAL
		CONSTRAINT PK_Genre PRIMARY KEY,
	Title VARCHAR(64) NOT NULL
);

CREATE TABLE Artist(
	ID SERIAL
		CONSTRAINT PK_Artist PRIMARY KEY,
	StageName VARCHAR(32),
	FirstName VARCHAR(32) NOT NULL,
	LastName VARCHAR(32) NOT NULL,
	Origin VARCHAR(32),
	Birthdate DATE
);

CREATE TABLE ArtistGenre(
	ArtistID INTEGER NOT NULL
		CONSTRAINT FK_ArtistGenre_Artist REFERENCES Artist(ID),
	GenreID INTEGER NOT NULL
		CONSTRAINT FK_ArtistGenre_Genre REFERENCES Genre(ID),
	CONSTRAINT UN_ArtistGenre UNIQUE(ArtistID,GenreID)
);

CREATE TABLE Band(
	ID SERIAL
		CONSTRAINT PK_Band PRIMARY KEY,
	Name VARCHAR(64) NOT NULL,
	Origin VARCHAR(64)
);

CREATE TABLE BandArtist(
	BandID INTEGER NOT NULL
		CONSTRAINT FK_BandArtist_Band REFERENCES Band(ID),
	ArtistID INTEGER NOT NULL
		CONSTRAINT FK_BandArtist_Artist REFERENCES Artist(ID),
	CONSTRAINT UN_BandArtist UNIQUE (BandID,ArtistID)
);

CREATE TABLE FooUser(
	ID SERIAL
		CONSTRAINT PK_FooUser PRIMARY KEY,
	UserName VARCHAR(32) NOT NULL
		CONSTRAINT UN_FooUser_UserName UNIQUE,
	FirstName VARCHAR(32) NOT NULL,
	LastName VARCHAR(32) NOT NULL,
	Email VARCHAR(32) NOT NULL,
	Password VARCHAR(32) NOT NULL
);

CREATE TABLE BandGenre(
	BandID INTEGER NOT NULL
		CONSTRAINT FK_BandGenre_Band REFERENCES Band(ID),
	GenreID INTEGER NOT NULL
		CONSTRAINT FK_BandGenre_Genre REFERENCES Genre(ID),
	CONSTRAINT UN_BandGenre UNIQUE (BandID,GenreID)
);

CREATE TABLE Album(
	ID SERIAL
		CONSTRAINT PK_Album PRIMARY KEY,
	Title VARCHAR(64) NOT NULL,
	ReleaseDate DATE
);

CREATE TABLE ArtistAlbum(
	ArtistID INTEGER
		CONSTRAINT FK_ArtistAlbum_Artist REFERENCES Artist(ID),
	AlbumID INTEGER NOT NULL
		CONSTRAINT FK_ArtistAlbum_Album REFERENCES Album(ID),
	CONSTRAINT UN_ArtistAlbum UNIQUE(ArtistID,AlbumID)
);

CREATE TABLE BandAlbum(
	BandID INTEGER NOT NULL
		CONSTRAINT FK_BandAlbum_Band REFERENCES Band(ID),
	AlbumID INTEGER NOT NULL
		CONSTRAINT FK_BandAlbum_Album REFERENCES Album(ID),
	CONSTRAINT UN_BandAlbum UNIQUE(BandID,AlbumID)
);

CREATE TABLE Track(
	ID SERIAL
		CONSTRAINT PK_Track PRIMARY KEY,
	Title VARCHAR(64) NOT NULL,
	Length REAL,
	AlbumID INTEGER
		CONSTRAINT FK_Track_Album REFERENCES Album(ID),
	GenreID INTEGER
		CONSTRAINT FK_Track_Genre REFERENCES Genre(ID)
);

CREATE TABLE TrackReview(
	ID SERIAL
		CONSTRAINT PK_TrackReview PRIMARY KEY,
    TrackID INTEGER NOT NULL
        CONSTRAINT FK_TrackReview_Track REFERENCES Track(ID)
        ON DELETE CASCADE,
    FooUserID INTEGER NOT NULL
        CONSTRAINT FK_TrackReview_FooUser REFERENCES FooUser(ID)
        ON DELETE CASCADE,
    Content TEXT NOT NULL
);

CREATE TABLE ArtistReview(
	ID SERIAL
		CONSTRAINT PK_ArtistReview PRIMARY KEY,
    ArtistID INTEGER NOT NULL
        CONSTRAINT FK_ArtistReview_Artist REFERENCES Artist(ID)
        ON DELETE CASCADE,
    FooUserID INTEGER NOT NULL
        CONSTRAINT FK_ArtistReview_FooUser REFERENCES FooUser(ID)
        ON DELETE CASCADE,
    Content TEXT NOT NULL
);

CREATE TABLE BandReview(
	ID SERIAL
		CONSTRAINT PK_BandReview PRIMARY KEY,
    BandID INTEGER NOT NULL
        CONSTRAINT FK_BandReview_Band REFERENCES Band(ID)
        ON DELETE CASCADE,
    FooUserID INTEGER NOT NULL
        CONSTRAINT FK_BandReview_FooUser REFERENCES FooUser(ID)
        ON DELETE CASCADE,
    Content TEXT NOT NULL
);

CREATE TABLE AlbumReview(
	ID SERIAL
		CONSTRAINT PK_AlbumReview PRIMARY KEY,
	AlbumID INTEGER NOT NULL
		CONSTRAINT FK_AlbumReview_Album REFERENCES Album(ID)
		ON DELETE CASCADE,
	FooUserID INTEGER NOT NULL
		CONSTRAINT FK_AlbumReview_FooUser REFERENCES FooUser(ID)
		ON DELETE CASCADE,
	Content TEXT NOT NULL
);

CREATE TABLE Playlist(
    ID SERIAL
        CONSTRAINT PK_Playlist PRIMARY KEY,
    Title VARCHAR(40),
    FooUserID INTEGER
        CONSTRAINT FK_Playlist_FooUser REFERENCES FooUser(ID)
        ON DELETE CASCADE
);

CREATE TABLE PlaylistTrack(
    TrackID INTEGER NOT NULL
        CONSTRAINT FK_PlaylistTrack_Track REFERENCES Track(ID)
        ON DELETE CASCADE,
    PlaylistID INTEGER NOT NULL
        CONSTRAINT FK_PlaylistTrack_Playlist REFERENCES Playlist(ID)
        ON DELETE CASCADE,
	CONSTRAINT UN_PlaylistTrack UNIQUE (TrackID, PlaylistID)
);

CREATE TABLE AlbumTrack(
    AlbumID INTEGER NOT NULL
        CONSTRAINT FK_AlbumTrack_Album REFERENCES Album(ID)
        ON DELETE CASCADE,
    TrackID INTEGER NOT NULL
        CONSTRAINT FK_AlbumTrack_Track REFERENCES Track(ID)
        ON DELETE CASCADE,
	CONSTRAINT UN_AlbumTrack UNIQUE (AlbumID, TrackID)
);
